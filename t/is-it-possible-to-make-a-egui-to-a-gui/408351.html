	private static JFrame frame;
	private static final long serialVersionUID = 1L;

	public Jframe(final String args[]) {

		try {
			sign.signlink.startpriv(InetAddress.getByName(client.server));
			this.initUI();
		} catch (final Exception ex) {

		}
	}

		try {

			}
		} catch (final Exception e) {

	@Override
	public URL getCodeBase() {
		try {
			return new URL("http://" + client.server + "/cache");
		} catch (final Exception e) {
			return super.getCodeBase();
		}
	}

	@Override
	public URL getDocumentBase() {
		return this.getCodeBase();
	}

	@Override
	public String getParameter(final String key) {
		return "";
	}

	public void initUI() {
		try {
			Jframe.frame = new JFrame("Client Name");
			Jframe.frame.setLayout(new BorderLayout());
			Jframe.frame.setResizable(false);
			Jframe.frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
			final JPanel gamePanel = new JPanel();
			gamePanel.setLayout(new BorderLayout());
			gamePanel.add(this);
			final Dimension dim = Toolkit.getDefaultToolkit().getScreenSize();
			final int w = 765;
			final int h = 503;
			final int x = (dim.width - w) / 2;
			final int y = (dim.height - h) / 2;
			Jframe.frame.setLocation(x, y);
			gamePanel.setPreferredSize(new Dimension(765, 503));
			final JMenu fileMenu = new JMenu("DropDown menu name here");
			final String[] mainButtons = new String[] { " Tool Bar Name here " };

			for (final String name : mainButtons) {
				final JMenuItem menuItem = new JMenuItem(name);
				menuItem.addActionListener(this);
				fileMenu.add(menuItem);
			}
			final JMenuBar menuBar = new JMenuBar();
			final JMenuBar jmenubar = new JMenuBar();

			Jframe.frame.add(jmenubar);
			menuBar.add(fileMenu);
			Jframe.frame.getContentPane().add(menuBar, BorderLayout.NORTH);
			Jframe.frame.getContentPane().add(gamePanel, BorderLayout.CENTER);
			Jframe.frame.pack();

			Jframe.frame.setVisible(true);
			Jframe.frame.setResizable(false);
			this.init();
		} catch (final Exception e) {
			e.printStackTrace();
		}
	}

	public void loadError(final String s) {
		System.out.println("loadError: " + s);
	}
}
