#3 - Method call counters
In all classes, there are counter variables at the beginning of each method.
Found no use of them yet, but they record how many times each method is called (all initialized in method50).
Useful for something like calculateScreenPosition, which can be used by the legit client and cheaters - too many calls and your user is cheating.

ok, here is my zkm log parser, it outputs it to an xml file which is easy as hell to open and get values from, the only thing that needs working on is storing/renaming the overridden methods (as it is now, if there are 5 methods named "a" in the class it will only store one of the new names) im welcoming suggestions on that, otherwise here it is(attached to post as well):

import java.io.*;
import java.util.Properties;
/**
 * Author: moparisthebest
 * <p/>
 * Converts zkm ChangeLog.txt into easy to use
 * xml format. Overridden methods still not
 * successfully handled.
 */
public class changeLogParser {
    public static int[] renamedAsKey = {1, 0};  //switch the zero and the one to switch key/content(duh?)
public Properties xmlDoc;
public int classes, fields, methods;


public static void main(String[] args) {
    new changeLogParser(args);
}

public changeLogParser(String[] args) {
    xmlDoc = new Properties();
    try {
        File file = new File("ChangeLog.txt");
        FileInputStream fistream = new FileInputStream(file);

        BufferedReader reader = new BufferedReader(new InputStreamReader(fistream));
        String line = "";
        String[] classNames = new String[2];
        while ((line = reader.readLine()) != null) {
            if (!line.contains("NameNotChanged"))
                if (line.contains("Class:")) {
                    classNames = getNames(line);
                    xmlDoc.setProperty(classNames[0], classNames[1]);
                    classes++;
                } else if ((!line.contains("(")) && (line.contains("=>"))) {
                    String [] fieldNames = getNames(line);
                    xmlDoc.setProperty(classNames[0] + "." + fieldNames[0], classNames[1] + "." + fieldNames[1]);
                    fields++;
                } else if ((line.contains("(")) && (line.contains("=>"))) {
                    String [] methodNames = getMethodNames(line);
                    xmlDoc.setProperty(classNames[0] + "." + methodNames[0], classNames[1] + "." + methodNames[1]);
                    methods++;
                }
        }

    } catch (Exception e) {
        e.printStackTrace();
    }
    if (writeToXML(xmlDoc, "ZKMchangeLog.xml", "zkm changeLog in xml format"))
        System.out.println("Successfully made ZKMchangeLog.xml\nZKM Renamed:\n" + classes + " Classes\n" + fields + " Fields\n" + methods + " Methods\n");
    else
        System.out.println("Error outputing ZKMchangeLog.xml");
}

public String[] getMethodNames(String line) {
    String[] tokens = line.split("\\s");
    String[] results = new String[2];
    for (String token : tokens)
        if (token.contains("("))
            results[renamedAsKey[0]] = token.substring(0, token.indexOf("("));
    results[renamedAsKey[1]] = tokens[tokens.length - 1];
    return results;
}

public String[] getNames(String line) {

    String[] tokens = line.split("\\s");
    String[] results = new String[2];
    results[renamedAsKey[0]] = tokens[tokens.length - 3];
    results[renamedAsKey[1]] = tokens[tokens.length - 1];
    return results;
}

public boolean writeToXML(Properties xmlDoc, String fileName, String comment) {
    try {
        File file = new File(fileName);
        FileOutputStream fos = new FileOutputStream(file);
        xmlDoc.storeToXML(fos, comment);
        return true;
    } catch (Exception e) {
        return false;
    }
}
}
