{Setup}
const
  Roads       =   7040116;  //Road color
  Trips       =   1      ;  // How many times to fight guards and die.
  Style       =   1      ;  // Style of fighting. Numbered from top to bottom
  Timer       =   20000  ;  // Maximum amount of time before clicking next
                            // Guard.
  StartPlayer =   0      ;  // Which player to start at.
  Gui         =  true    ;  // Use it once at true, then set to false
  

{Setup end}

Const
  Vrr         =  '1.2'   ;






{========================VAR======================================}

procedure SwapPlays(Reason:String; State:Boolean); // Little thingy
begin
  Writeln('Switched because of '+ Reason);
  Nextplayer(State);
end;

{--------------------------------------------------------------------------}


procedure RandomTab;
var  ac, at:Integer;
begin
 ac:=1+random(50)
     if(ac<=7)then
     begin
          GameTab(ac);
          case ac of
            1: SetFightMode(1+random(3))
            2: hoverskill('random');
            4: UseItem(2+random(27))
          end;
          wait(3000+random(2000));
          at:= 1+random(8);
          GameTab(at);
          wait(3000+random(2000));
     end;
end;


{--------------------------------------------------------------------------}

procedure Mo;
begin
  getClientDimensions(x, y);
  MMouse(Random(x), Random(y), 2, 2);
end;


procedure EquipItems;
var
  ABCD:Integer;
begin
  for ABCD:= 1 to 3 do
    MouseItem(ABCD, true)
end;

{--------------------------------------------------------------------------}


{--------------------------------------------------------------------------}

procedure RightRandClick;
var
  PColor, TColor, RColor: Integer;
  begin
    case Random(5) of               // Random PlayerColor
      0: PColor:= 7708622;
      1: PColor:= 6786229;
      2: PColor:= 6983609;
      3: PColor:= 7774416;
      4: PColor:= 4221833;
    end;
    case Random(3) of                 // Random Plant Color
      0: TColor:= 1251352;
      1: TColor:= 2056280;
      2: TColor:= 3362888;
    end;
    RColor:= 6184806;                 // Road color
    begin
      case random(10) of
        3: begin
             if (FindMainColor(x, y, PColor, 80, false)) then
               begin
                 Mouse(x, y, 2, 2, false);
                 wait(1000+Random(200));
               end;
           end;
        5: begin
             if (FindMainColor(x, y, TColor, 80, false)) then
               begin
                 Mouse(x, y, 2, 2, false);
                 wait(1000+Random(200));
               end;
           end;
        8: begin
             if (FindMainColor(x, y, RColor, 80, false)) then
               begin
                 Mouse(x, y, 2, 2, false);
                 wait(1000+Random(200));
               end;
           end;
      end;
    end;
  end;
  
  

{--------------------------------------------------------------------------}


procedure R;
begin
  case Random(10) of
    1: RandomTab;
    2: Mo;
    4: RightRandClick;
  end;
end;

{--------------------------------------------------------------------------}


procedure R2;
begin
  case Random(10) of
    1: RandomTab;
    2: Mo;
    5: RightRandClick;
  end;
end;



{--------------------------------------------------------------------------}




procedure WalkToBlackWizards;// by fakawi (<3 <3 <3 ,<3) edited by me
begin
  RadialWalk(Rosh,90,0,65,-1,1);
  wait(1)
  R;
  FindnormalRandoms;
  RadialWalk(Rosh,90,0,65,-1,1);
  wait(1)
  R;
  FindnormalRandoms;
  RadialWalk(Rosh,0,90,65,0,1);
  MarkTime(Mark);
  repeat
    RadialWalk(Rosh,420,250,35,0,1);
    FindnormalRandoms;
  until( (FindSymbol(x, y, 'cook')) or (TimeFromMark(Mark)>600000) );
end;

{--------------------------------------------------------------------------}

procedure WalktoCastle;  // walks to varrock castle
begin
  repeat
    LinearWalk(Rosh, 0, 72, -1, 1);
    FindNormalRandoms
  until(Findsymbol(x, y, 'water'))
  Players[CurrentPlayer].Loc := 'Varrock Castle';
  Mouse(x, y, 1, 1, true)
end;


{--------------------------------------------------------------------------}
procedure Sttack(Color1, color2, color3, color4, color5, color6, color7:Integer; Name:String; Tolerance, Tij:Integer);// Made by fakawi, edited by me a bit.
var
  Bx, GColor: Integer;
begin
  for Bx := 1 to 7 do
  begin
   case Bx of
    1: GColor:= Color1;
    2: GColor:= Color2;
    3: GColor:= Color3;
    4: GColor:= Color4;
    5: GColor:= Color5;
    6: GColor:= Color6;
    7: GColor:= Color7;
  end;

  if (not( Loggedin)) then break;
  begin
     if (GColor=0) then break;
     if (FindObj(x, y, 'ttack '+Name, GColor, tolerance)) then
     begin
        GetMousePos(x, y);
        if(not (Loggedin) or (WeAreDead)) then break;
        Mouse(x, y, 0, 0, True);
        if(not (Loggedin) or (WeAreDead)) then break;
        wait(2000);
        FFlag(10);
        RandomTab;
        MarkTime(Mark);
        repeat
           if(not (Loggedin) or (WeAreDead)) then break;
           wait(1000+Random(500));
           if (TimeFromMark(Mark)>Tij) then break;
        until ( (not (FindColorSpiral(x,y,GreenStatusColor,240,140,290,175)) or (WeAreDead) ));
        Killed:=Killed+1
     end
     else
     begin
       writeln('restart in different world')
       TerminateScript;
     end;
   end
 end
end;

procedure FindNorRandoms;   // Edited to my use
var
  i : Integer;
begin
 for i:=1 to 15 do
  begin
  case I of
   1:  CloseWindow;
   2:  FindSandwich;
   3:  FindLamp(LampSkill);
   4:  FindMod
   5:  FindMime;
   6:  FindMaze;
   7:  Findquiz;
   8:  FindFrog;
   9:  FindDemon;
   10: FindScapeRune;
   11: BoxTime;
   12: FindTrade;
   13: begin
        if(not(Loggedin))then
           Exit;
       end;
   14: begin
        if(NoGameTab)then
         begin
           Players[CurrentPlayer].loc := 'No GameTab';
           Logout;
          Exit;
         end;
       end;
   15: R;
  end;
  wait(1);
 end;
end;

procedure ProgressReport;
var
  Hours, Minutes, Seconds: Integer;
  Time:String;
begin
  ClearDebug;
  Hours:= ScriptTime(1);
  Minutes:= ScriptTime(2);
  Seconds:= ScriptTime(3);
  Time:= inttostr(Hours)+ ' Hours, ' +inttostr(Minutes)+ ' Minutes, and ' +inttostr(Seconds)+ ' Seconds.';
  WriteLn('[-----------------------------------------------]');
  WriteLn('Script ran for '+Time                             );
  Writeln('Player:=       '+Players[CurrentPlayer].Name      );
  WriteLn('Made '+inttostr(died)+' time[s]'                  );
  WriteLn('Killed '+inttostr(killed)+' guard[s]'             );
  WriteLn(' Thanks for using a SESAME script.   '            );
  writeLn('  __   __                            '            );
  Writeln(' /   |/  |      / /                  '            );
  Writeln('(___ |   | ___   (___                '            );

  Writeln('__/  |__/ |__  | |__                 '            );
  WriteLn('[-----------------------------------------------]');
end;

procedure Loop;  //Well... does everything
begin
{Sets up}
  SetupSRL;
  BenMouse:= true;
  Reincarnate:=true;
  if (Gui) then
    begin
      SetupPlayers;
    end else
    LoadUS;
  CurrentPlayer:=StartPlayer;
  if (not (Loggedin)) then
    LoginPlayer;
  repeat
    Killed:=0
    SetChat('off', 1);
    SetChat('off', 2);
    SetChat('off', 3);
    SetChat('off', 1); // makes sure
    NickNameBMP:=CreateBitmapMaskFromText(Players[CurrentPlayer].Nick,UpChars);
    Died:=0;
    Swit:=false;
    Swit2:=false;
    PerfectNorth;
    Rosh:= Roads;
{/Sets up}

     repeat //Does the killing
{Walks}
      if (Players[CurrentPlayer].Loc = 'Lumbridge') then
        begin
          WalkToBlackWizards; //Fakwi
          FindNormalRandoms;
          WalktoCastle;// my little thingy
        end;
{/walks}



{The guards are being lazy, this is unacceptable, go teach them a lesson}
      if (Players[CurrentPlayer].Loc = 'Varrock Castle') then
        begin
          repeat
            AddtoReport('Killing guards')
            TrainGuards;
            R2;
            FindnorRandoms;
            if wearedead then begin Died:= Died+1; Players[CurrentPlayer].Loc:='Lumbridge'; break; EquipItems end;
            if (Died = Trips) then begin Swit:=true; break; end;
            if (not (loggedin)) then begin Swit2:=true; Break;end;
          until (false)
        end;
{/The guards are being lazy, this is unacceptable, go teach them a lesson}

    until Swit or Swit2;

{Switch?}
    begin
      if (Swit2) then
        SwapPlays('For a random reason your out',False);
        ProgressReport;
      if (Swit) then
        ProgressReport;
        SwapPlays(Players[currentplayer].name+' is done with '+inttostr(Trips)+' trips', true);
   end;
   if (not (Loggedin)) then  NextPlayer(False);  // Always the last line!
  until false

{/Switch?}
end;

{--------------------------------------------------------------------------}
      
begin
  ChangeReportWidth(0)
  Loop;
end.
