|A basic script for lv 3-9 |
|Doesnt collect feathers |
|Doesnt colllect meat |
|Doesnt collect bones|
|I like fried chicken :P|

program New;
{.include SRL/SRL.scar}
{.include SRL/SRL/Skill/Fighting.scar}
//MADE BY WALL OF FIR



const
chicken=924541; // this is the chickens color(try get the color of their red beaks)

procedure setup;
begin
setupsrl
end;

procedure kill;
begin
clickmscolor(chicken,true);
wait(3000);
OutFight
end;

procedure bob;
begin
FindNormalRandoms;
end;


begin
setup;
repeat
kill;
bob;
until(false)
end.[/scar]
