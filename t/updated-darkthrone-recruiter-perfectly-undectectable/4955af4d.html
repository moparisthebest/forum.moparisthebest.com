Ron July 31, 2016, 8:23am #100

This works fine on scar 3.20d as of June 2009.

program DTRecruiterByRon;

var
  dtmRecruit,
  x, y, x2, y2,
  i : integer;

procedure Load;
begin
  dtmRecruit := DTMFromString('78DA63FCC4C8C0D00FC448C0CC510C4CC3441' +
              '98F0159ADA86AF64EF2465523CFC4C0300355CDE62E174C3568E6' +
              '1C98ED8169179A396BDB9C50D4000078C30C5C');
end;

procedure Free;
begin
  FreeDTM( dtmRecruit );
end;

begin
  ClearDebug;
  GetClientDimensions( x2, y2 );
  if not( FindWindowTitlePart( 'Dark Throne Recruit', False ) ) then
  begin
    WriteLn( 'Cannot find Dark Throne Recruit window.' );
    TerminateScript;
  end else
    ActivateClient;
  Load;
  i := 0;
  while not( IsFKeyDown( 12 ) ) do
  begin
    if ( FindDTM( dtmRecruit, x, y, 0, 0, x2, y2 ) ) then
    begin
      x := x + StrToInt(FloatToStr(Pow(-1, Random(2)))) * Random( 40 );
      y := y + StrToInt(FloatToStr(Pow(-1, Random(2)))) * Random( 10 );
      MoveMouseSmooth( x, y );
      HoldMouse( x, y, True );
      Wait( 10 + Random( 20 ) );
      ReleaseMouse( x, y, True );
      inc( i );
      WriteLn('Recruited ' + IntToStr( i ) + ' citizen(s).');
    end;
    Wait( 1000 + Random( 2000 ) );
  end;
  Free;
end.
