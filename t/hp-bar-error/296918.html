private static void appendHit2Update(Player p, StaticPacketBuilder updateBlock) {
if(p.getHits().getHitDamage2() < 128) {
updateBlock.addByte((byte) p.getHits().getHitDamage2());
} else {
updateBlock.addShort(p.getHits().getHitDamage2() + 32768);
}
updateBlock.addByte((byte) p.getHits().getHitType2());
}

private static void appendHitUpdate(Player p, StaticPacketBuilder updateBlock) {
	if(p.getHits().getHitDamage1() < 128) {
		updateBlock.addByte((byte) p.getHits().getHitDamage1());
	} else {
		updateBlock.addShort(p.getHits().getHitDamage1() + 32768);
	}
	updateBlock.addByteC(p.getHits().getHitType1());
	int hpRatio = p.getSkills().getLevel(3) * 255 / p.getSkills().getLevelForXp(3);
	updateBlock.addByte((byte) hpRatio);
