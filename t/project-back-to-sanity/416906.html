EDIT : moved the whole Client.process() method to the new GameTick class.
It should be called straight from the Server class. (take note of the other minigames and such that are handled the same way)
The GameTick class goes in the Server.model.players package.

GameTick class:

[code=java]package server.model.players;

import server.model.players.Client;
import server.Config;
import server.Server;
import server.util.Misc;
import server.util.Stream;
import server.model.players.PlayerHandler;

public class GameTick {

/**
	@author Cres
*/

public GameTick() {
}

public void tick() {
	/*
	Get all the players registered
	*/
	for (int j = 0; j < Server.playerHandler.players.length; j++) {
		/*
		Nullcheck on all the clients
		*/
		if (Server.playerHandler.players[j] != null) {
			Client c = (Client)Server.playerHandler.players[j];
			//TODO stuff goes here

			if (System.currentTimeMillis() - c.specDelay > Config.INCREASE_SPECIAL_AMOUNT) {
				c.specDelay = System.currentTimeMillis();
				if(c.specAmount < 10) {
						c.specAmount += .5;
					if (c.specAmount > 10)
						c.specAmount = 10;
					c.getItems().addSpecialBar(c.playerEquipment[c.playerWeapon]);
				}
			}

			if (c.clickObjectType > 0 && c.goodDistance(c.objectX + c.objectXOffset, c.objectY + c.objectYOffset, c.getX(), c.getY(), c.objectDistance)) {
				if (c.clickObjectType == 1) {
					c.getActions().firstClickObject(c.objectId, c.objectX, c.objectY);
				}
				if (c.clickObjectType == 2) {
					c.getActions().secondClickObject(c.objectId, c.objectX, c.objectY);
				}
				if (c.clickObjectType == 3) {
					c.getActions().thirdClickObject(c.objectId, c.objectX, c.objectY);
				}
			}

			if ((c.clickNpcType > 0) && Server.npcHandler.npcs[c.npcClickIndex] != null) {		
				if (c.goodDistance(c.getX(), c.getY(), Server.npcHandler.npcs[c.npcClickIndex].getX(), Server.npcHandler.npcs[c.npcClickIndex].getY(), 1)) {
					if (c.clickNpcType == 1) {
						c.turnPlayerTo(Server.npcHandler.npcs[c.npcClickIndex].getX(), Server.npcHandler.npcs[c.npcClickIndex].getY());
						Server.npcHandler.npcs[c.npcClickIndex].facePlayer(c.playerId);
						c.getActions().firstClickNpc(c.npcType);
					}
					if (c.clickNpcType == 2) {
						c.turnPlayerTo(Server.npcHandler.npcs[c.npcClickIndex].getX(), Server.npcHandler.npcs[c.npcClickIndex].getY());
						Server.npcHandler.npcs[c.npcClickIndex].facePlayer(c.playerId);
						c.getActions().secondClickNpc(c.npcType);
					}
					if (c.clickNpcType == 3) {
						c.turnPlayerTo(Server.npcHandler.npcs[c.npcClickIndex].getX(), Server.npcHandler.npcs[c.npcClickIndex].getY());
						Server.npcHandler.npcs[c.npcClickIndex].facePlayer(c.playerId);
						c.getActions().thirdClickNpc(c.npcType);
					}
				}
			}
			
			if (c.walkingToItem) {
				if (c.getX() == c.pItemX && c.getY() == c.pItemY || c.goodDistance(c.getX(), c.getY(), c.pItemX, c.pItemY,1)) {
					c.walkingToItem = false;
					Server.itemHandler.removeGroundItem(c, c.pItemId, c.pItemX, c.pItemY, true);
				}
			}
			
			if (c.followId > 0) {
				c.getPA().followPlayer();
			} else if (c.followId2 > 0) {
				c.getPA().followNpc();
			}
			
			c.getCombat().handlePrayerDrain();
			
			if(System.currentTimeMillis() - c.singleCombatDelay >  3300) {
				c.underAttackBy = 0;
			}
			
			if (System.currentTimeMillis() - c.singleCombatDelay2 > 3300) {
				c.underAttackBy2 = 0;
			}
			
			if (System.currentTimeMillis() - c.restoreStatsDelay >  60000) {
				c.restoreStatsDelay = System.currentTimeMillis();
				for (int level = 0; level < c.playerLevel.length; level++)  {
					if (c.playerLevel[level] < c.getLevelForXP(c.playerXP[level])) {
						if (level != 5) { // prayer doesn't restore
							c.playerLevel[level] += 1;
							c.getPA().setSkillLevel(level, c.playerLevel[level], c.playerXP[level]);
							c.getPA().refreshSkill(level);
						}
					} else if (c.playerLevel[level] > c.getLevelForXP(c.playerXP[level])) {
						c.playerLevel[level] -= 1;
						c.getPA().setSkillLevel(level, c.playerLevel[level], c.playerXP[level]);
						c.getPA().refreshSkill(level);
					}
				}
			}
			
			if (System.currentTimeMillis() - c.teleGrabDelay >  1550 && c.usingMagic) {
				c.usingMagic = false;
				if(Server.itemHandler.itemExists(c.teleGrabItem, c.teleGrabX, c.teleGrabY)) {
					Server.itemHandler.removeGroundItem(c, c.teleGrabItem, c.teleGrabX, c.teleGrabY, true);
				}
			}
			
			if (c.inWild()) {
				int modY = c.absY > 6400 ?  c.absY - 6400 : c.absY;
				c.wildLevel = (((modY - 3520) / 8) + 1);
				c.getPA().walkableInterface(197);
				if (Config.SINGLE_AND_MULTI_ZONES) {
					if (c.inMulti()) {
						c.getPA().sendFrame126("@yel@Level: "+c.wildLevel, 199);
					} else {
						c.getPA().sendFrame126("@yel@Level: "+c.wildLevel, 199);
					}
				} else {
					c.getPA().multiWay(-1);
					c.getPA().sendFrame126("@yel@Level: "+c.wildLevel, 199);
				}
				c.getPA().showOption(3, 0, "Attack", 1);
			} else if (c.inDuelArena()) {
				c.getPA().walkableInterface(201);
				if (c.duelStatus == 5) {
					c.getPA().showOption(3, 0, "Attack", 1);
				} else {
					c.getPA().showOption(3, 0, "Challenge", 1);
				}
			} else if (c.inBarrows()){
				c.getPA().sendFrame99(2);
				c.getPA().sendFrame126("Kill Count: "+c.barrowsKillCount, 4536);
				c.getPA().walkableInterface(4535);
			} else if (c.inCwGame || c.inPits) {
				c.getPA().showOption(3, 0, "Attack", 1);	
			} else if (c.getPA().inPitsWait()) {
				c.getPA().showOption(3, 0, "Null", 1);
			} else if (!c.inCwWait) {
				c.getPA().sendFrame99(0);
				c.getPA().walkableInterface(-1);
				c.getPA().showOption(3, 0, "Null", 1);
			}
			
			if (!c.hasMultiSign && c.inMulti()) {
				c.hasMultiSign = true;
				c.getPA().multiWay(1);
			}
			
			if (c.hasMultiSign && !c.inMulti()) {
				c.hasMultiSign = false;
				c.getPA().multiWay(-1);
			}

			if (c.skullTimer > 0) {
				c.skullTimer--;
				if (c.skullTimer == 1) {
					c.isSkulled = false;
					c.attackedPlayers.clear();
					c.headIconPk = -1;
					c.skullTimer = -1;
					c.getPA().requestUpdates();
				}	
			}
			
			if (c.isDead && c.respawnTimer == -6) {
				c.getPA().applyDead();
			}
			
			if (c.respawnTimer == 7) {
				c.respawnTimer = -6;
				c.getPA().giveLife();
			} else if (c.respawnTimer == 12) {
				c.respawnTimer--;
				c.startAnimation(0x900);
				c.poisonDamage = -1;
			}	
			
			if (c.respawnTimer > -6) {
				c.respawnTimer--;
			}
			
			if (c.freezeTimer > -6) {
				c.freezeTimer--;
				if (c.frozenBy > 0) {
					if (Server.playerHandler.players[c.frozenBy] == null) {
						c.freezeTimer = -1;
						c.frozenBy = -1;
					} else if (!c.goodDistance(c.absX, c.absY, Server.playerHandler.players[c.frozenBy].absX, Server.playerHandler.players[c.frozenBy].absY, 20)) {
						c.freezeTimer = -1;
						c.frozenBy = -1;
					}
				}
			}
			
			if(c.hitDelay > 0) {
				c.hitDelay--;
			}
			
			if (c.teleTimer > 0) {
				c.teleTimer--;
				if (!c.isDead) {
					if (c.teleTimer == 1 && c.newLocation > 0) {
						c.teleTimer = 0;
						c.getPA().changeLocation();
					}
					if (c.teleTimer == 5) {
						c.teleTimer--;
						c.getPA().processTeleport();
					}
					if (c.teleTimer == 9 && c.teleGfx > 0) {
						c.teleTimer--;
						c.gfx100(c.teleGfx);
					}
				} else {
					c.teleTimer = 0;
				}
			}	

			if (c.hitDelay == 1) {
				if (c.oldNpcIndex > 0) {
					c.getCombat().delayedHit(c.oldNpcIndex);
				}
				if (c.oldPlayerIndex > 0) {
					c.getCombat().playerDelayedHit(c.oldPlayerIndex);				
				}		
			}
			
			if (c.attackTimer > 0) {
				c.attackTimer--;
			}
			
			if (c.attackTimer == 1){
				if (c.npcIndex > 0 && c.clickNpcType == 0) {
					c.getCombat().attackNpc(c.npcIndex);
				}
				if (c.playerIndex > 0) {
					c.getCombat().attackPlayer(c.playerIndex);
				}
			} else if (c.attackTimer <= 0 && (c.npcIndex > 0 || c.playerIndex > 0)) {
				if (c.npcIndex > 0) {
					c.attackTimer = 0;
					c.getCombat().attackNpc(c.npcIndex);
				} else if (c.playerIndex > 0) {
					c.attackTimer = 0;
					c.getCombat().attackPlayer(c.playerIndex);
				}
			}
			
			if (c.timeOutCounter > Config.TIMEOUT) {
				c.disconnected = true;
			}
			
			c.timeOutCounter++;
			
			if (c.inTrade && c.tradeResetNeeded){
				Client o = (Client) Server.playerHandler.players[c.tradeWith];
				if(o != null){
					if(o.tradeResetNeeded){
						c.getTradeAndDuel().resetTrade();
						o.getTradeAndDuel().resetTrade();
					}
				}
			}
		}
	}
}

}[/code]
