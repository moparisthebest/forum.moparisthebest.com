import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.DataInputStream;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.io.PrintWriter;

import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.List;

public class Donator {
    public List<String> donatorList;
    public BufferedReader reader;

    public Donator() {
    	donatorList = new ArrayList<String>();
    	refreshDonatorList();
    }
    
    public boolean isDonator(String playerName) {
        return donatorList.contains(playerName);
    }
    
    public void refreshDonatorList() {
    	donatorList.clear();
    	try {
    		DataInputStream in = new DataInputStream(new FileInputStream("./data/Donators.txt"));
    		BufferedReader br = new BufferedReader(new InputStreamReader(in));
    	    
    		String strLine;
    		while ((strLine = br.readLine()) != null)   {
    			donatorList.add(strLine);
    		}
    		in.close();
    	} catch (FileNotFoundException e){
    		System.err.println("File Not Found, make sure ./data/Donators.txt exists.");
    	} catch(IOException e) {
    		System.err.println("Error: " + e.getMessage());
    	}
    }
    
    public void addDonator(String playerName) {
    	try {
                BufferedWriter bw = new BufferedWriter(new FileWriter("./data/Donators.txt", true));
                bw.write(playerName);
                bw.close();
			
                refreshDonatorList();
    	} catch (IOException e) {
	        e.printStackTrace();
	}
    }
    
    public void removeDonator(String playerName) {
    	try {
    		File inFile = new File("./data/Donators.txt");
    		File tempFile = new File(inFile.getAbsolutePath() + ".tmp");
    		      
    		BufferedReader br = new BufferedReader(new FileReader("./data/Donators.txt"));
    		PrintWriter pw = new PrintWriter(new FileWriter(tempFile));
    		      
    		String line = null;
    		while ((line = br.readLine()) != null) {
    			if (!line.trim().equals(playerName)) {
    				pw.println(line);
    		        pw.flush();
    		    }
    		}
    		pw.close();
    		br.close();
    		      
    		inFile.delete();
    		tempFile.renameTo(inFile);
    		
    		refreshDonatorList();
    	} catch (FileNotFoundException ex) {
    		ex.printStackTrace();
    	} catch (IOException ex) {
    		ex.printStackTrace();
    	}
    }
}
