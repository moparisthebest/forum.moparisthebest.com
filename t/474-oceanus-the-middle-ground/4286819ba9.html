package org.hyperion.rs2.model.skills.crafting;

import org.hyperion.rs2.action.impl.ProductionAction;
import org.hyperion.rs2.model.Animation;
import org.hyperion.rs2.model.Graphic;
import org.hyperion.rs2.model.Item;
import org.hyperion.rs2.model.ItemDefinition;
import org.hyperion.rs2.model.Mob;
import org.hyperion.rs2.model.Skills;

/**
 * Handles the leather crafting etc
 * 
 * @author Ares_
 * 
 * Date Created: 1/19/2012
 * 
 * Last Edit: 1/19/2012
 */
public class Crafting extends ProductionAction {

    /**
     * The amount of items to produce.
     */
    private int productionCount;
        
    /**
     * The leather we are crafting
     */
    private Leather leather;
    
    /**
     * The leather index
     */
    private int leatherIndex;
    
    public Crafting(Mob mob, int productionCount, int leatherIndex, Leather leather) {
        super(mob);
        this.productionCount = productionCount;
        this.leatherIndex = leatherIndex;
        this.leather = leather;
    }
    
    @Override
    public int getCycleCount() {
        return 3;
    }

    @Override
    public int getProductionCount() {
        return productionCount;
    }

    @Override
    public Item[] getRewards() {
        //return new Item[] { new Item(log.getItem()[logIndex], log.getItem()[logIndex] == 53 ? 15 : 1) };
        return new Item[] { new Item(leather.getReward()[leatherIndex], leather.getReward()[leatherIndex] == 53 ? 15 : 1) };
    }

    @Override
    public Item[] getConsumedItems() {
        return new Item[] { new Item(leather.getHideId()) };
    }

    @Override
    public int getSkill() {
        return Skills.CRAFTING;
    }

    @Override
    public int getRequiredLevel() {
        return leather.getLevelRequired()[leatherIndex];
    }

    @Override
    public double getExperience() {
        return leather.getExperience()[leatherIndex];
    }

    @Override
    public String getLevelTooLowMessage() {
        return "You need a Crafting level of " + getRequiredLevel() + " to craft this.";
    }

    @Override
    public String getSuccessfulProductionMessage() {
        String prefix = "a";
        String suffix = "";
        char first = ItemDefinition.forId(leather.getReward()[leatherIndex]).getName().toLowerCase().charAt(0);
        if(first == 'a' || first == 'e' || first == 'i' || first == 'o' || first == 'u') {
            prefix = "an";
	}
        return "You successfully craft " + prefix + " " + ItemDefinition.forId(leather.getReward()[leatherIndex]).getName().toLowerCase() + "" + suffix + ".";
    }

    @Override
    public Animation getAnimation() {
        return Animation.create(1249);
    }

    @Override
    public Graphic getGraphic() {
        return null;
    }

    @Override
    public boolean canProduce() {
        return true;
    }
}

The leather enum

package org.hyperion.rs2.model.skills.crafting;

import java.util.HashMap;
import java.util.Map;

/**
 * The leather data
 * 
 * @author Ares_
 * 
 * Date Created: 1/19/2012
 *
 * Last Edit: 1/19/2012
 */
public enum Leather {
    BLACK_HIDE(2509, new int[] {2503, 2491, 2497}, new int[] {3, 1, 2}, new int[] {84, 79, 82}, new double[] {258.0, 86.0, 172.0});
     
    /**
     * A map of item ids to leathers.
     */
    private static Map<Integer, Leather> leathers = new HashMap<Integer, Leather>();
    
    Leather(int hideId, int[] reward, int[] amounts, int[] levelRequired, double[] experience) {
        this.hideId = hideId;
        this.reward = reward;
        this.amounts = amounts;
        this.levelRequired = levelRequired;
        this.experience = experience;
    }
    
    public static Leather forId(int hide) { return leathers.get(hide); }
    
    /**
     * Populates the map
     */
    static {
        for(Leather hide : Leather.values()) {
           leathers.put(hide.hideId, hide);
        }
    }
    
    /**
     * The hide required
     */
    private int hideId;
    
    /**
     * The rewards you gain
     */
    private int[] reward;
    
    /**
     * The required for each thing needing to be created
     */
    private int[] amounts;
    
    /**
     * The level you need to craft that item
     */
    private int[] levelRequired;
    
    /**
     * The experience gained from crafting items
     */
    private double[] experience;
    
    /**
     * @return the hide id
     */
    public int getHideId() {
        return hideId;
    }
    
    /**
     * @param index
     * @return the product rewarded
     */
    public int[] getReward() {
        return reward;
    }
    
    /**
     * @param index
     * @return the amount of hides required
     */
    public int[] getAmounts() {
        return amounts;
    }
    
    /**
     * @param index
     * @return the level needed to craft a certain leather piece
     */
    public int[] getLevelRequired() {
        return levelRequired;
    }
    
    /**
     * @param index
     * @return the experience gained from crafting a leather piece
     */
    public double[] getExperience() {
        return experience;
    }
}
