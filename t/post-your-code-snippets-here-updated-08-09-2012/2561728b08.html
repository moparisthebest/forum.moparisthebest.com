	 * Checks a player for a specific item. Also checks amount.
	 * 
	 * @param item
	 *            The item in question.
	 * @return Whether or not the player has the item in their bank, in their inventory, or on their person.
	 */
	public boolean checkForItem(Item item) {
		if (player.getInventory().getItemContainer().containsItem(item)) {
			return true;
		}
		
		if (player.getBank().containsItem(item)) {
			return true;
		}
		if (player.getEquipment().getItemContainer().containsItem(item)) {
			return true;
		}
		
		return false;
	}

Hyperion npc spawning via a cfg

I do not recommend this, I wrote it awhile back, I would now suggest using a database or through XML. 

[B][I]package org.hyperion.rs2.model;[/I][/B]

add this class
[code]
/**
 * @Author Broham No documentation as this is extremely old and I couldn`t be
 *         bothered to do so.
 **/
public class NPCSpawn {

	public static ArrayList<SpawnData> spawnList = new ArrayList<SpawnData>();

	private static final Logger logger = Logger.getLogger(NPCSpawn.class
			.getName());

	public static void loadSpawn() {

		BufferedReader file = null;
		logger.info("Loading spawns");
		try {
			file = new BufferedReader(new FileReader("./data/spawns.cfg"));
			while (true) {
				String line = file.readLine();
				if (line == null)
					break;
				int spot = line.indexOf('=');
				if (spot > -1) {
					String values = line.substring(spot + 1);
					values = values.replace("\t\t", "\t");
					values = values.replace("\t\t", "\t");
					values = values.trim();
					String[] valuesArray = values.split("\t");

					int id = Integer.valueOf(valuesArray[0]);

					spawnList.add(new SpawnData(id, Location.create(
							Integer.valueOf(valuesArray[1]),
							Integer.valueOf(valuesArray[2]), 0)));
					addNPC(Integer.valueOf(valuesArray[1]),
							Integer.valueOf(valuesArray[2]), 0, id);
				}
			}
			logger.info("Loaded " + World.getWorld().getNPCs().size()
					+ " NPC spawns.");
		} catch (IOException ioe) {
			ioe.printStackTrace();
		} finally {
			if (file != null)
				try {
					file.close();
				} catch (IOException e) {
					e.printStackTrace();
				}
		}
	}

	public static void addNPC(int x, int y, int z, int npcId) {
		NPCDefinition nD = new NPCDefinition(npcId);
		NPC n = new NPC(nD);
		n.setLocation(Location.create(x, y, z));
		Region region = World.getWorld().getRegionManager()
				.getRegionByLocation(n.getLocation());
		region.addNpc(n);
		World.getWorld().register(n);
	}

	public static class SpawnData {
		private int id;
		private Location loc;

		public SpawnData(int id, Location loc) {
			this.setId(id);
			this.loc = loc;
		}

		public int getId() {
			return id;
		}

		public Location getLoc() {
			return loc;
		}

		public void setId(int id) {
			this.id = id;
		}
	}
}
[/code]

/**
 * <p>Represents a type of NPC.</p>
 * @author Graham Edgecombe
 *
 */
public class NPCDefinition {
	
	/**
	 * Gets an npc definition by its id.
	 * @param id The id.
	 * @return The definition.
	 */
	public static NPCDefinition forId(int id) {
		return new NPCDefinition(id);
	}
	
	/**
	 * The id.
	 */
	private int id;
	
	/**
	 * Creates the definition.
	 * @param id The id.
	 */
	NPCDefinition(int id) {
		this.id = id;
	}
	
	/**
	 * Gets the id.
	 * @return The id.
	 */
	public int getId() {
		return this.id;
	}

}
