Program SimonSolver;

Const
  LevelToStopAt    = 0;     // Leave at 0 to go to level infinity!
  TimesToCheck     = 50;    // 50 is default. Increase if you see any errors.
  NotifyWithSounds = true;  // If true, this will notify you with sounds.
  ShowDebugInfo    = false; // If true, this will show 'extra' info in the debug box.

  // Wait inbetween detecting colors and clicking. I used events for this,
  // so I don't know why I can't just get rid of the wait. But if you do
  // that then it will detect the color more than once. Go figure. :(
  WaitT = 500; // 500 is the lowest you can go without getting errors later.

  // Wait inbetween clicking on each color. I used events for this, so
  // I don't know why I can't just get rid of the wait. But if you do that
  // then the script clicks too fast. Go figure. :(
  WaitT2 = 500; // 500 is the lowest you can go without getting errors later.

  // When an error occurs play this sound. Set NotifyWithSounds to false
  // to turn this off. Default is 'C:\Windows\Media\chord.wav'.
  SoundError = 'C:\Windows\Media\chord.wav';

  // When you press F12 to stop the script, or the LevelToStopAt
  // has been reached, play this sound. Set NotifyWithSounds to false to
  // turn this off. Default is 'C:\Windows\Media\tada.wav'.
  SoundCongrats = 'C:\Windows\Media\tada.wav';

// Declare variables.
Var
  // Miscellaneous variables.
  x, y, x2, y2, x3, y3, x4, y4, x5, y5, i, n, s, m, w, z, P2, B : Integer;
  // Bitmap variables.
  Play, Quit, Level, Red, Blue, Green, Yellow, Simon, GameOver : Integer;
  // A few strings.
  P, P3, P4, P5, P6, Msg, Version : String;

// Load all bitmaps.
Procedure LoadBMPs;
Begin
  // Play button.
  Play     := BitmapFromString(3, 3, 'z78DA7375730542087045622303570C' +
              '350095030EB9');
  // Quit button.

       'FAFAF');
  // Red button lit.
  Red      := BitmapFromString(3, 3, 'FFC2C2FFC0C0FFBFBFFFBFBFFFBEBEF' +
              'FBCBCFFBCBCFFBABAFFB8B8');
  // Blue button lit.
  Blue     := BitmapFromString(3, 3, '9393FF9191FF8E8EFF9494FF9393FF9' +
              '090FF9696FF9393FF9292FF');
  // Green button lit.
  Green    := BitmapFromString(3, 3, 'B5FFB5B3FFB3B2FFB2B3FFB3B1FFB1B' +
              '0FFB0B1FFB1AFFFAFAEFFAE');
  // Yellow button lit.
  Yellow   := BitmapFromString(3, 3, 'FFFFF9FFFFF5FFFFF3FFFFF5FFFFF3F' +
              'FFFF0FFFFF2FFFFF0FFFFED');
  // Simon button lit.
  Simon    := BitmapFromString(2, 2, '7A7A7A6E6E6E8F8F8F7E7E7E');
  // Game over.
  GameOver := BitmapFromString(2, 2, '68686873736181815995954F');
  WriteLn('All bitmaps loaded.');
End;

// Free all bitmaps.
Procedure FreeBMPs;
Begin
  FreeBitmap(Play);
  FreeBitmap(Quit);
  FreeBitmap(Red);
  FreeBitmap(Blue);
  FreeBitmap(Green);
  FreeBitmap(Yellow);
  FreeBitmap(Simon);
  FreeBitmap(GameOver);
  WriteLn('All bitmaps freed.');
End;

// Detect GameOver.
Procedure DetectGameOver;
Begin
  // If the GameOver bitmap is found then...
  If(FindBitmap(GameOver, x, y))Then
  Begin
    // If NotifyWithSounds is true then...
    If(NotifyWithSounds)Then
    Begin
      // Play sound error.
      PlaySound(SoundError);
    End;
    // Display an error popup message.
    GetApplication.MESSAGEBOX('The script screwed up and a Game Over was detected.','Simon Solver '+Version+' by Ron',0);
    // Free all the bitmaps.
    FreeBMPs;
    // Kill the script.
    TerminateScript;
  End;
End;

// Writes the progress report.
Procedure ProgressReport;
Begin
  // Clears the debug box.
  ClearDebug;
  // Clears the report box.
  ClearReport;
  // Writes the progress report in the report box.
  AddToReport('Simon Solver '+Version+' by Ron');
  AddToReport('  Level: '+IntToStr(Level));
  AddToReport('  Score: '+IntToStr(Level-1));
  // If ShowDebugInfo is true then...
  If(ShowDebugInfo)Then
  Begin
    AddToReport('  Simons Found: '+IntToStr(s)+'/'+IntToStr(s+m));
    AddToReport('  PatternBefore: '+P6);
    AddToReport('  PatternAdded: '+P5);
  End;
  AddToReport('  PatternResult: '+P);
  AddToReport('  Broke Out: '+IntToStr(B))
End;

// Procedure NotFindBitmap to notify you when it can't find a certain bmp.
// 1 = Red, 2 = Blue, 3 = Green, 4 = Yellow.
Procedure NotFindBitmap(Number : Integer);
Begin
  // Message used later in the script.
  Msg := ' detected, continuing with script.';
  // Case of 1 keep repeating until Red isn't found.
  Case Number of
  1 : Begin
        Repeat
          Wait(5);
        Until(Not(FindBitmap(Red, x2, y2)))
        If(ShowDebugInfo)Then
        Begin
          WriteLn(IntToStr(n)+'. Red'+Msg);
        End;
      End;
  // Case of 2 keep repeating until Blue isn't found.
  2 : Begin
        Repeat
          Wait(5);
        Until(Not(FindBitmap(Blue, x3, y3)))
        If(ShowDebugInfo)Then
        Begin
          WriteLn(IntToStr(n)+'. Blue'+Msg);
        End;
      End;
  // Case of 3 keep repeating until Green isn't found.
  3 : Begin
        Repeat
          Wait(5);
        Until(Not(FindBitmap(Green, x4, y4)))
        If(ShowDebugInfo)Then
        Begin
          WriteLn(IntToStr(n)+'. Green'+Msg);
        End;
      End;
  // Case of 4 keep repeating until Yellow isn't found.
  4 : Begin
        Repeat
          Wait(5);
        Until(Not(FindBitmap(Yellow, x5, y5)))
        If(ShowDebugInfo)Then
        Begin
          WriteLn(IntToStr(n)+'. Yellow'+Msg);
        End;
      End;
   End;
End;

// This procedure fixed a bug where one of the colors was not detected.
// This goes around that problem by making 2 temporary strings called P4 and
// P5. Then as colors are added to P4, P5 gets the last letter of P4, and
// adds it to P. P is safe with all the colors and it runs perfect!
Procedure CheckStrings;
Begin
  // Get the length of P4 and store in to z.
  z := Length(P4);
  // Get the last letter of P4 and temporarily put it into P5.
  P5 := Copy(P4, z, 1);
  // If ShowDebugInfo is true then...
  If(ShowDebugInfo)Then
  Begin
    WriteLn(P+' + '+P5+' = ');
  End;
  // Set P6 to the original P.
  P6 := P;
  // Adds P5 to P.
  P := P6 + P5;
  // If ShowDebugInfo is true then...
  If(ShowDebugInfo)Then
  Begin
    WriteLn(P);
  End;
End;

// If the color = 'R' 'B' 'G' or 'Y' then click on the corresponding bmp.
Procedure Find(Color : String);
Begin
  If(Color = 'R')Then
  Begin
    // Click on Red.
    ClickMouse(x2, y2, True);
    // Adds 1 to n.
    n := n + 1;
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      WriteLn(IntToStr(n)+'. Clicked on Red.');
    End;
    // Keeps checking to see if the bitmap isn't found. When it isn't,
    // it continues.
    NotFindBitmap(1);
  End Else
  If(Color = 'B')Then
  Begin
    // Click on Blue.
    ClickMouse(x3, y3, True);
    // Adds 1 to n.
    n := n + 1;
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      WriteLn(IntToStr(n)+'. Clicked on Blue.');
    End;
    // Keeps checking to see if the bitmap isn't found. When it isn't,
    // it continues.
    NotFindBitmap(2);
  End Else
  If(Color = 'G')Then
  Begin
    // Click on Green.
    ClickMouse(x4, y4, True);
    // Adds 1 to n.
    n := n + 1;
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      WriteLn(IntToStr(n)+'. Clicked on Green.');
    End;
    // Keeps checking to see if the bitmap isn't found. When it isn't,
    // it continues.
    NotFindBitmap(3);
  End Else
  If(Color = 'Y')Then
  Begin
    // Click on Yellow.
    ClickMouse(x5, y5, True);
    // Adds 1 to n.
    n := n + 1;
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      WriteLn(IntToStr(n)+'. Clicked on Yellow.');
    End;
    // Keeps checking to see if the bitmap isn't found. When it isn't,
    // it continues.
    NotFindBitmap(4);
  End Else
  Begin
    // Error report.
    WriteLn('Error: Color wasn''t either R, B, G, or Y!');
    // Free all bitmaps.
    FreeBMPs;
    // Kill the script.
    TerminateScript;
  End;
  // Detect a Game Over.
  DetectGameOver;
End;

// Check for Simon lit up.
Procedure CheckForSimon;
Begin
  // If faded bitmap is not found then...
  If(Not(FindBitmap(Simon, x, y)))Then
  Begin
    // Adds 1 to s.
    s := s + 1;
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      WriteLn('Simon Found!');
    End;
  // Otherwise...
  End Else
  Begin
    // Adds 1 to m.
    m := m + 1;
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      WriteLn('Simon Not Found!');
    End;
  End;
End;

// This finds the Play button and clicks on it.
Procedure Start;
Begin
  Level := 0;
  Version := 'v.31';
  // If the Play bitmap is found then...
  If(FindBitmap(Play, x, y))Then
  Begin
    // Click on it.
    ClickMouse(x, y, True);
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      // Display that the play button was clicked and the game started.
      WriteLn('Play button was clicked. Game Started.');
    End;
  // Otherwise...
  End Else
  If(FindBitmap(Quit, x, y))Then
  Begin
    // Click on it.
    ClickMouse(x, y, True);
    // If ShowDebugInfo is true then...
    If(ShowDebugInfo)Then
    Begin
      // Display that the play button was clicked and the game started.
      WriteLn('Quit button detected and clicked.');
    End;
    Wait(50);
    // If bitmap Play is found then...
    If(FindBitmap(Play, x, y))Then
    Begin
      // Click on it.
      ClickMouse(x, y, True);
      // If ShowDebugInfo is true then...
      If(ShowDebugInfo)Then
      Begin
        // Display that the play button was clicked and the game started.
        WriteLn('Play button detected and clicked. Game Started.');
      End;
    End Else
    Begin
      // Display an error.
      WriteLn('Error: Play button not detected.');
      // Free bitmaps.
      FreeBMPs;
      // Kills the script.
      TerminateScript;
    End;
  End Else
  Begin
    // Display the error.
    WriteLn('Error: Play button nor Quit button detected.');
    // Free bitmaps.
    FreeBMPs;
    // Kills the script.
    TerminateScript;
  End;
End;

// If the script does not find the image then it will add that image that
// it could not find to the string P.
Procedure Remember;
Begin
  // Clear temporary variables.
  P4 := '';
  P5 := '';
  w := 0;
  n := 0;
  z := 0;
  Repeat
    // If it finds the bitmap, save it to the variables x2, y2, and then...
    If(FindBitmap(Red, x2, y2))Then
    Begin
      // Adds the first letter of the color it found.
      P4 := P4 + 'R';
      // Adds 1 to n.
      n := n + 1;
      // And don't continue until the bitmap we just recorded is gone.
      NotFindBitmap(1);
      // Set w back to 0.
      w := 0;
    // Otherwise...
    End Else
    // If it finds the bitmap, save it to the variables x3, y3, and then...
    If(FindBitmap(Blue, x3, y3))Then
    Begin
      // Adds the first letter of the color it found.
      P4 := P4 + 'B';
      // Adds 1 to n.
      n := n + 1;
      // And don't continue until the bitmap we just recorded is gone.
      NotFindBitmap(2);
      // Set w back to 0.
      w := 0;
    // Otherwise...
    End Else
    // If it finds the bitmap, save it to the variables x4, y4, and then...
    If(FindBitmap(Green, x4, y4))Then
    Begin
      // Adds the first letter of the color it found.
      P4 := P4 + 'G';
      // Adds 1 to n.
      n := n + 1;
      // And don't continue until the bitmap we just recorded is gone.
      NotFindBitmap(3);
      // Set w back to 0.
      w := 0;
    // Otherwise...
    End Else
    // If it finds the bitmap, save it to the variables x5, y5, and then...
    If(FindBitmap(Yellow, x5, y5))Then
    Begin
      // Adds the first letter of the color it found.
      P4 := P4 + 'Y';
      // Adds 1 to n.
      n := n + 1;
      // And don't continue until the bitmap we just recorded is gone.
      NotFindBitmap(4);
      // Set w back to 0.
      w := 0;
    End Else
    // If w is greater than or equal to TimesToCheck then...
    If(w >= TimesToCheck)Then
    Begin
      // Display error.
      WriteLn('Error: Script hung and didn''t catch one of the colors.');
      WriteLn('breaking from loop.');
      // Adds 1 to B.
      B := B + 1;
      // If NotifyWithSounds is true then...
      If(NotifyWithSounds)Then
      Begin
        // Play SoundError.
        PlaySound(SoundError);
      End;
      // Break out of the loop.
      Break;
    End;
    // Adds 1 to w.
    w := w + 1;
    // Changes Status and shows how many times it checked.
    Status('Checked for colors '+IntToStr(w)+'/'+IntToStr(TimesToCheck)+' time(s).');
  // Keep repeating until n is equal to Level.
  Until(Level = n)
  // If ShowDebugInfo is true then...
  If(ShowDebugInfo)Then
  Begin
    WriteLn('--------------------------------------------------');
  End;
  // Set status back to 'Running Script...'
  Status('Running Script...');
  // Checks strings for errors and adds the last letter of the temporary
  // string, to the real pattern string. :)
  CheckStrings;
  // Small wait before starting the click process.
  Wait(WaitT);
End;

// Click the pattern.
Procedure ClickPattern;
Begin
  // Gets the length of P.
  P2 := Length(P);
  // Set n back to 0.
  n := 0;
  // i is 1 and i to the length of p do this...
  For i := 1 to P2 do
  Begin
    // Set P3 to the letter
    P3 := Copy(P, i, 1);
    // Find P3. This will send the letter the function Copy() gave to P3.
    // Then it will find the letter, color, and click on it.
    Find(P3);
    // Changes status and shows it's clicking.
    Status('Clicked on colors '+IntToStr(i)+'/'+IntToStr(P2)+' time(s).');
    // Small wait.
    Wait(WaitT2);
  End;
  // Checks for the lit up Simon.
  CheckForSimon;
  // Set status back to 'Running Script...'
  Status('Running Script...');
End;

Begin
  // Clear debug box.
  ClearDebug;
  // Clear report box.
  ClearReport;
  // Changes report box's width.
  ChangeReportWidth(230);
  // Load all the bitmaps.
  LoadBMPs;
  // Activate the client.
  ActivateClient;
  // Start by clicking on the button play and setting some variables.
  Start;
  // Repeat until F12 is pushed down or until LevelToStopAt equals Level.
  Repeat
    // Move mouse away from the screen.
    MoveMouse(0, 0);
    // Adds one to the Level.
    Level := Level + 1;
    // Record the colors.
    Remember;
    // Click on the colors.
    ClickPattern;
    // Print progress report.
    ProgressReport;
  Until(IsFKeyDown(12)) or (LevelToStopAt = Level)
  // If the level to stop at is greater than 0 then...
  If(LevelToStopAt > 0)Then
  Begin
    // Display a congrats message.
    WriteLn('Level '+IntToStr(Level)+' reached!');
  End;
  // Plays a tada sound when finished.
  If(NotifyWithSounds)Then
  Begin
    PlaySound(SoundCongrats);
  End;
  // Free the bitmaps.
  FreeBMPs;
End.
