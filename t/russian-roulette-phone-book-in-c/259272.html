// Function to delete an entry.
EntryPointer deleteEntry (EntryPointer head, char last[])
{
// Declaration of variables
EntryPointer current = head, previous = NULL;
bool found = false;

// If input matches a current entry, found is set to true.
while (!found && current != NULL)
    if (strcmp(current->last, last) == 0)
        found = true;
    else
    {
        // Otherwise, keep on looking.
        previous = current;
        current = current->link;
    }

if (found)
    if (current == head)
        head = current->link;
    else
        previous->link = current->link;

return head;

}

EntryPointer deleteList (EntryPointer head)
{
EntryPointer next;
while (head != NULL)
{
next = head->link;
free(head);
head = next;
}
return NULL;

EntryPointer insertInOrder (EntryPointer head, char last[], char first[], char address[], char num[])
{
EntryPointer curr = head, prev = NULL;
bool located = false;

while (!located && curr != NULL)
    if (strcmp(last, curr->last) < 0)
        located = true;
    else
    {
        prev = curr;
        curr = curr->link;
    }

EntryPointer newItem = newEntry(last, first, address, num, curr);

if (curr == head)
    head = newItem;
else
    prev->link = newItem;

return head;

}
