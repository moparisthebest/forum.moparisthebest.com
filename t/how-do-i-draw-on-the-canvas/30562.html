Pretty simple:

[scar]program New;
const
width = 300;
height = 300;
var
dbg,buffer: TCanvas;
bmp: Integer;
begin
//Make dummy bitmap and get the canvas (since SCAR can‚Äôt just create 'em, lol)
bmp:= BitmapFromString(width,height,#0);
buffer:= GetBitmapCanvas(bmp);
//Display it and get handle
DisplayDebugImgWindow(width,height);
dbg:= GetDebugCanvas;

//Draw to buffer then copy to debug canvas
buffer.Brush.Color:= clWhite;
buffer.Pen.Color := clRed;
buffer.Rectangle(10,10,width-10,height-10);
CopyCanvas(buffer,dbg,0,0,width,height,0,0,width,height);
end.[/scar]
Buffer usually only needed when drawing lots of things.
