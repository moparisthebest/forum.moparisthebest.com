[scar]
//[============================================]
//[ To be used with Trapshoot game, ]
//[ http://www.dragongamez.com/trapshoot.htm ]
//[ Created by:Aznrx ]
//[============================================]
//[ - Go to website ]
//[ - Drag crosshair onto game ]
//[ - Select any character ]
//[ - Select any target you want, hit run ]
//[ - F6 ingame for Chickens ]
//[ - F7 ingame for Cans ]
//[ - F8 ingame for Clay pigeons ]
//[ - Note: You need to press spacebar ingame ]
//[============================================]
//[ This is my first ever script! ]
//[ Thanks to sythe.org/moparisthebest.com ]
//[ for its helpful info. This script was ]
//[ tested by me. Enjoy =D ]
//[============================================]\

program Trapshoot;
var
x,y:integer;

procedure shootchicken;
begin
repeat
if(findcolor(x,y,7384045,10, 11,540, 337)=True)then
begin
clickmouse(x,y,True);
wait(300)
end;
until(false);
end;

procedure shootcan;
begin
repeat
if(findcolor(x,y,13260,10, 11,540, 337)=True)then
begin
clickmouse(x,y,True);
wait(400)
end;
until(false);
end;

procedure shootclay;
begin
repeat
if(findcolor(x,y,1078746,10, 11,540, 337)=True)then
begin
clickmouse(x,y,True);
wait(400)
end;
until(false);
end;

begin
repeat
if isfkeydown(6)=true then shootchicken;
if isfkeydown(7)=true then shootcan;
if isfkeydown(8)=true then shootclay;
wait(100)
until(false)
end.[/scar]

a simple scar script for the trapshoot game.
