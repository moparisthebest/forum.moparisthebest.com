       @echo off
                  title SAR(Server Auto Restarter)
           :start
                  set time=1800
                  start run.bat
           :loop
                  cls
                  IF %time% GTR 0 (
                   set /a time=%time% - 1
                    set /a min=%time%/60
                     echo 
                    echo   %time% Until Restart %min%.
                   ping 127.0.0.1 -n 2 > NUL
                  goto loop
                  ) 
           taskkill /f /im java.exe
          cls
         goto start
