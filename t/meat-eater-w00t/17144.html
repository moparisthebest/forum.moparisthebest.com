[ Have a different type of cooked meat in last, 2nd ]
[ last and 3rd last for auto color. ]
[ Declare how many different types of meat you are ]
[ eating on line 18. ]
[ Drag crosshair onto runescape. ]
[ Press green arrow. ]
[=====================================================]
[Made for cow meat but could work on others. ]
[=====================================================}
program WallMeatEater;
var x,y,cookedmeat,cookedmeat2,cookedmeat3:integer;
const
i = 1; //number of meat types

procedure GetMeatColor;
begin
cookedmeat :=getcolor(713,445);
cookedmeat2:=getcolor(668,445);
cookedmeat3:=getcolor(625,445);
end;

procedure EatMeat;
begin
repeat
if(FindColor(x,y,cookedmeat, 534, 171, 759, 497)) then
movemousesmoothEx(x,y,30,30,30,17,30);
sleep(50);
clickmouse(x,y,true);
sleep(1900);
until(not(FindColor(x,y,cookedmeat, 534, 171, 759, 497)))
end;

procedure EatMeat2;
begin
if(i=2) then
begin
repeat
if(FindColor(x,y,cookedmeat2, 534, 171, 759, 497)) then
movemousesmoothEx(x,y,30,30,30,17,30);
sleep(50);
clickmouse(x,y,true);
sleep(1900);
until(not(FindColor(x,y,cookedmeat2, 534, 171, 759, 497)))
end else
if(not(FindColor(x,y,cookedmeat2, 534, 171, 759, 497))) then
begin
terminatescript;
end;
end;

procedure EatMeat3;
begin
if(i=3) then
begin
repeat
if(FindColor(x,y,cookedmeat3, 534, 171, 759, 497)) then
movemousesmoothEx(x,y,30,30,30,17,30);
sleep(50);
clickmouse(x,y,true);
sleep(1900);
until(not(FindColor(x,y,cookedmeat3, 534, 171, 759, 497)))
end else
if(not(FindColor(x,y,cookedmeat3, 534, 171, 759, 497))) then
begin
terminatescript;
end;
end;

begin
activateclient;
GetMeatColor;
EatMeat;
EatMeat2;
EatMeat3;
end.[/scar]
