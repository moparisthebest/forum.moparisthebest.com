[code]program New;

const
UseWait = True; //True for Yes, False for No
WaitTime = 5; //Seconds to Wait between Clicks

{
SCRIPT USER: DO NOT EDIT BELOW THIS LINE
}

var
WaitFor: Integer;

begin
WaitFor := WaitTime * 1000;
repeat
MouseBox(5, 5, 5, 5);
Wait(WaitFor);
until(IsFKeyDown(2)); //Always have a failsafe, in this case, Press F2 to stop
end;[/code]

begin if UseWait = True then begin WaitFor := WaitTime * 1000; repeat MouseBox(5, 5, 5, 5); Wait(WaitFor); until(IsFKeyDown(2)); //Always have a failsafe, in this case, Press F2 to stop end else Writeln('Wait is off'); end.
