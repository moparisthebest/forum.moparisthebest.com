Are you on in edgevillepvp source? This is how I fixed mine:

			if (System.currentTimeMillis() - c.foodDelay < 2000)
			return;
			c.foodDelay = System.currentTimeMillis();
			if(c.eatemote==0){c.setAnimation(0x33D);}
			//c.playSound(317, 000, 020);
			c.sM("You eat shark.");	
			if(c.currentHealth < 99) {
			c.currentHealth += 20;
			c.sendQuest("" + c.currentHealth + "", 4016);
			}
			if(c.currentHealth > 99) {
			c.currentHealth = 99;
			}
			c.deleteItem(385, slot, 1);
			break;
